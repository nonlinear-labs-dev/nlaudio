101 inline void sinewave(T *buffer, float frequency, bool reset, const SampleSpecs_t& sampleSpecs)
102 {
103  static T scale = std::numeric_limits<T>::is_signed ?
104  ((1 << (sampleSpecs.bytesPerSample * 8)) / 2) :
105  (1 << (sampleSpecs.bytesPerSample * 8));
106 
107  static T offset = !std::numeric_limits<T>::is_signed ?
108  (1 << (sampleSpecs.bytesPerSample * 8) / 2) :
109  0;
110 
111  static float phase = 0.f;
112  static float inc = frequency / static_cast<float>(sampleSpecs.samplerate);
113 
114  if (reset) {
115  phase = 0.f;
116  inc = frequency / static_cast<float>(sampleSpecs.samplerate);
117  }
118 
119  for (unsigned int i=0; i<sampleSpecs.buffersizeInFramesPerPeriode; i++) {
120  phase += inc;
121 
122  if (phase > 0.5)
123  phase -= 1.0;
124 
125 #if 1
126  float x = 2 * phase;
127 
128  if (x < 0) {
129  x = -x;
130  }
131 
132  x = 0.5 - x;
133 
134  float x_square = x * x;
135  x = x * (x_square * (x_square * 2.26548 - 5.13274) + 3.14159);
136  buffer[i] = static_cast<T>(x * scale + offset);
137 #else
138  buffer[i] = sin(2.f * M_PI * phase) * scale + offset;
139 #endif
140  }
